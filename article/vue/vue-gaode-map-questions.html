      this.addDrawingPolyline([position, position]) // æ·»åŠ ç»˜åˆ¶è¿‡ç¨‹ä¸­çš„é¼ æ ‡ç§»åŠ¨ä½ç½®çš„æŠ˜çº¿
      this.drawPolygonByPoints() // é€šè¿‡ç‚¹å›´æˆå¤šè¾¹å½¢
    },
    // åœ°å›¾åŒå‡»äº‹ä»¶
    mapOnDblclick (ev) {
      if (!this.isDrawing) return
      this.complateDraw()
    },
    // åœ°å›¾é¼ æ ‡ç§»åŠ¨äº‹ä»¶
    mapOnMouseMove (ev) {
      if (!this.isDrawing) return
      const position = [ev.lnglat.lng, ev.lnglat.lat]
      const linePath = this.getPointLine(position) // è·å–ç‚¹æ‰€åœ¨çš„çº¿
      let recentPoint = null // å®šä¹‰æœ€è¿‘çš„ç‚¹
      if (this.drawingPolygon.isOnPolygon) {
        // å¦‚æœç‚¹åœ¨çº¿ä¸Šï¼ˆè¿™ä¸ªç‚¹å­˜åœ¨åç§»ï¼Œæ‰€ä»¥è¿˜å¾—é€šè¿‡getRecentPointè·å–åˆ°æ­£åœ¨çš„æœ€è¿‘çš„ä¸€ä¸ªç‚¹ï¼‰
        // è·å–è¯¥ç‚¹åˆ°çº¿çš„æœ€è¿‘çš„ä¸€ä¸ªç‚¹
        recentPoint = this.getRecentPoint(linePath, position)
        this.addDrawingOnLinePoint(recentPoint) // åœ¨çº¿ä¸Šæ·»åŠ åœ†ç‚¹ï¼Œä¹Ÿå°±æ˜¯è‡ªåŠ¨å¸é™„çš„æ—¶å€™åˆ›å»ºçš„ç‚¹
      } else {
        this.removeDrawingOnLinePoint()
      }
      this.setDrawingPolyline(position)
    },
    // æ·»åŠ å›´æˆå¤šè¾¹å½¢çš„ç‚¹
    addPolygonPoint (position) {
      // æ ·å¼åŠåæ ‡option
      const option = {
        ...this.styles.circleMarker,
        center: position
      }
      const circlePointMarker = new AMap.CircleMarker(option) // å•ä¸ªåœ†ç‚¹å®ä¾‹
      // å…ˆåˆ¤æ–­æ˜¯å¦å­˜åœ¨åœ†ç‚¹é›†åˆå®ä¾‹
      // æ‰€æœ‰çš„åœ†ç‚¹è¿™é‡Œé€šè¿‡é«˜å¾·æä¾›çš„OverlayGroupç»Ÿä¸€ç®¡ç†
      if (!this.drawingPolygon.pointsGroup) { // å¦‚æœä¸å­˜åœ¨
        this.drawingPolygon.pointsGroup = new AMap.OverlayGroup() // åˆ›å»ºç»˜åˆ¶è¿‡ç¨‹ä¸­ç‚¹çš„é›†åˆçš„å®ä¾‹
        this.GDMap.add(this.drawingPolygon.pointsGroup) // å°†é›†åˆæ·»åŠ åˆ°åœ°å›¾ä¸Šæ˜¾ç¤º
      }
      this.drawingPolygon.pointsGroup.addOverlay(circlePointMarker) // å°†ç‚¹æ·»åŠ åˆ°é›†åˆé‡Œé¢
    },
    // æ–°å¢ç»˜åˆ¶ä¸­çš„æŠ˜çº¿
    addDrawingPolyline (paths) {
      // åˆ¤æ–­æœ‰æ— æŠ˜çº¿
      if (this.drawingPolygon.polyline) {
        this.drawingPolygon.polyline.setPath(paths) // æœ‰çš„è¯ç›´æ¥è®¾ç½®æŠ˜çº¿è·¯å¾„
      } else { // æ²¡æœ‰çš„è¯éœ€è¦åˆ›å»ºæŠ˜çº¿
        // æŠ˜çº¿æ ·å¼åŠè·¯å¾„
        const option = {
          ...this.styles.drawingPolyline,
          path: paths
        }
        this.drawingPolygon.polyline = new AMap.Polyline(option) // ç”ŸæˆæŠ˜çº¿
        this.GDMap.add(this.drawingPolygon.polyline) // åœ°å›¾ä¸Šæ·»åŠ æŠ˜çº¿
      }
      this.drawingPolygon.polylinePath = paths // å­˜ä¸€ä¸‹æŠ˜çº¿çš„è·¯å¾„
    },
    // è®¾ç½®ç»˜åˆ¶ä¸­çš„æŠ˜çº¿çš„è·¯å¾„
    setDrawingPolyline (position) {
      if (this.drawingPolygon.polyline) {
        // æ–°å¢çš„æŠ˜çº¿çš„è·¯å¾„ï¼Œç”±ä¸Šæ¬¡è®°å½•æŠ˜çº¿çš„å¼€å§‹ç‚¹+é¼ æ ‡åæ ‡ä½ç½®ç‚¹
        const linePath = [
          this.drawingPolygon.polylinePath[0], // ä¸Šæ¬¡è®°å½•æŠ˜çº¿çš„å¼€å§‹ç‚¹
          position // é¼ æ ‡åæ ‡ä½ç½®ç‚¹
        ]
        this.drawingPolygon.polyline.setPath(linePath)
        this.drawingPolygon.polylinePath = linePath
      }
    },
    // æ ¹æ®ç‚¹ç”»åŒºå—å¤šè¾¹å½¢
    drawPolygonByPoints () {
      const pointsGroup = this.drawingPolygon.pointsGroup // ç‚¹åŠçš„é›†åˆå®ä¾‹
      const pointsLength = pointsGroup ? pointsGroup.getOverlays().length : 0 // ç‚¹çš„é•¿åº¦
      if (pointsLength > 1) {
        // è·å–æ¯ä¸ªç‚¹çš„ä¸­å¿ƒç‚¹ç»„æˆpath
        const paths = pointsGroup.getOverlays().map(item => {
          const path = item.getCenter()
          return [path.lng, path.lat]
        })
        // åœ°å›¾ä¸Šç»˜åˆ¶å¤šè¾¹å½¢
        if (this.drawingPolygon.polygon) {
          this.drawingPolygon.polygon.setPath(paths)
        } else {
          const option = {
            ...this.styles.polygon,
            path: paths
          }
          this.drawingPolygon.polygon = new AMap.Polygon(option)
          this.drawingPolygon.polygon.setMap(this.GDMap)
        }
      }
    },
    // ç»˜åˆ¶å®Œæˆ
    complateDraw () {
      const paths = this.drawingPolygon.polygon.getPath().map(item => [item.lng, item.lat]) // ç»˜åˆ¶å®Œæˆçš„å¤šè¾¹å½¢path
      this.isDrawing = false // å–æ¶ˆç»˜åˆ¶
      this.polygons.push(paths) // æ·»åŠ ç»˜åˆ¶çš„å¤šè¾¹å½¢
      this.addPolygon(paths) // åœ°å›¾ä¸Šæ·»åŠ ç»˜åˆ¶çš„å¤šè¾¹å½¢
      this.clearUselessOverlays() // æ¸…ç†æ•°æ®
    },
    // æ¸…ç†æ— ç”¨çš„å›¾å±‚
    clearUselessOverlays () {
      if (!this.GDMap) return
      this.GDMap.remove(this.drawingPolygon.polyline)
      this.GDMap.remove(this.drawingPolygon.polygon)
      this.GDMap.remove(this.drawingPolygon.pointsGroup)
      if (this.drawingPolygon.pointOnLine) {
        this.GDMap.remove(this.drawingPolygon.pointOnLine)
      }
      this.drawingPolygon.polyline = null
      this.drawingPolygon.polygon = null
      this.drawingPolygon.pointsGroup = null
      this.drawingPolygon.polylinePath = []
      this.drawingPolygon.isOnPolygon = false
    },
    // æ·»åŠ å¤šè¾¹å½¢
    addPolygon (paths) {
      if (!this.GDMap) return
      const option = {
        ...this.styles.polygon,
        path: [...paths]
      }
      const polygon = new AMap.Polygon(option)
      if (!this.polygonsGroup) {
        this.polygonsGroup = new AMap.OverlayGroup()
        this.GDMap.add(this.polygonsGroup)
      }
      this.polygonsGroup.addOverlay(polygon)
    },
    // æ·»åŠ å¤šè¾¹å½¢è¾¹çº¿ä¸Šçš„ç‚¹
    addDrawingOnLinePoint (center) {
      if (this.drawingPolygon.pointOnLine) {
        this.drawingPolygon.pointOnLine.setCenter(center)
        return
      }
      const option = {
        map: this.GDMap,
        center: center,
        ...this.styles.circleMarker
      }
      this.drawingPolygon.pointOnLine = new AMap.CircleMarker(option)
    },
    // ç§»é™¤å¤šè¾¹å½¢çº¿ä¸Šçš„ç‚¹
    removeDrawingOnLinePoint () {
      if (this.drawingPolygon.pointOnLine) {
        this.GDMap.remove(this.drawingPolygon.pointOnLine)
        this.drawingPolygon.pointOnLine = null
      }
    },
    // è·å–ç‚¹æ‰€åœ¨çš„çº¿
    getPointLine (position) {
      const resolution = this.GDMap.getResolution()// è·å–æŒ‡å®šä½ç½®çš„åœ°å›¾åˆ†è¾¨ç‡ï¼Œå•ä½ï¼šç±³/åƒç´ 
      const pointWidth = 6 * resolution// çº¿æ®µä¸Šåœ†ç‚¹çš„å®½åº¦ï¼Œä¹Ÿå°±æ˜¯è¯¯å·®
      let linePath = []
      this.drawingPolygon.isOnPolygon = false // é»˜è®¤ç‚¹ä¸åœ¨çº¿ä¸Š
      // å¾ªç¯æ‰€æœ‰çš„å¤šè¾¹å½¢ï¼Œå–åˆ°æ‰€æœ‰çš„çº¿ä¸€ä¸€æ¯”è¾ƒï¼Œå­˜åœ¨æ€§èƒ½é—®é¢˜
      for (let i = 0; i < this.polygons.length; i++) {
        const itemPath = this.polygons[i]
        let hasFind = false
        for (let n = 0; n < itemPath.length; n++) {
          const path = itemPath[n]
          const nextPath = itemPath[n + 1]
          const line = nextPath ? [path, nextPath] : [path, itemPath[0]]
          // é«˜å¾·åœ°å›¾æä¾›çš„isPointOnSegmentåˆ¤æ–­ç‚¹æ˜¯å¦åœ¨çº¿æ®µä¸Š
          const isPointOnSegment = AMap.GeometryUtil.isPointOnSegment(position, line[0], line[1], pointWidth)
          if (isPointOnSegment) {
            linePath = line
            this.drawingPolygon.isOnPolygon = true
            hasFind = true
            break
          }
        }
        if (hasFind) break
      }
      return linePath
    },
    // è·å–æœ€è¿‘çš„ç‚¹çš„åæ ‡
    getRecentPoint (paths, curPointPosition) {
      const recentPoint = AMap.GeometryUtil.closestOnLine(curPointPosition, paths)
      return recentPoint
    }
  },
  watch: {
    isDrawing (newState) {
      if (newState) {
        this.GDMap.setDefaultCursor('crosshair')
      } else {
        this.GDMap.setDefaultCursor('')
      }
      this.GDMap.setStatus({
        doubleClickZoom: !newState
      })
    }
  }
}
